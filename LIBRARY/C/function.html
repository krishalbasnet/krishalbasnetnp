1. WAP to reverse a number entered by user using the concept of function.
#include<stdio.h>
void main(){
    int n, rev=0, rem;
    printf("Enter a number: ");
    scanf("%d", &n);
    rev = reverse(n);
    printf("Reverse of %d is %d", n, rev);
}
int reverse(int n){
    int rev=0, rem;
    while(n!=0){
        rem = n%10;
        rev = rev*10 + rem;
        n = n/10;
    }
    return rev;
}
