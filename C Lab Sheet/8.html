LAB SHEET NO.8 - To be familiar with Pointers

1. WAP to find the sum and difference of two number using pointers

    #include<stdio.h>
    void main(){
      int a,b,sum,*x,*y;
      x=&a;
      y=&b;
      printf("Enter the value of a and b: ");
      scanf("%d%d",&a,&b);
      printf("The sum is %d",*x+*y);
    }
OUTPUT:-
Enter the value of a and b: 2 3
The sum is 5

2. WAP to read two values from the user in main(). Write a function calculate() to calculate sum, product, difference and quotient of the numbers. Print the sum, product, difference and quotient in the main function using pass by reference concept.

    #include<stdio.h>
    void calculate(int a, int b,int *p, int *q,int *r, float *s){
      *p = a + b;
      *q = a - b;
      *r = a * b;
      *s = (float)a / b;
    }
    void main(){
      int a,b,sum,sub,mul;
      float div;
      printf("Enter two numbers: ");
      scanf("%d%d",&a,&b);
      calculate(a,b,&sum,&sub,&mul,&div);
      printf("Sum:%d\tDifference:%d\tProduct:%d\tQuotient:%f",sum,sub,mul,div);
    }
OUTPUT:-
Enter two numbers: 1 2
Sum:3 Difference:-1 Product:2 Quotient:0.500000

3. WAP to access the elements of the array using the pointers.

    #include <stdio.h>
    int main() {
        int data[5];
        printf("Enter elements: ");
        for (int i = 0; i < 5; ++i)
            scanf("%d", data + i);
        printf("You entered: \n");
        for (int i = 0; i < 5; ++i)
            printf("%d\n", *(data + i));
        return 0;
    }
OUTPUT:-
Enter elements: 1 2 3 4 5
You entered:
1
2
3
4
5

4. WAP to illustrate the concept of array of pointers and pointer to the array.

    #include <stdio.h>
    void main() {
        int arr[] = {1, 2, 3, 4, 5};
        int* ptrArr[5];
        int(*ptrToArr)[5];           
        for (int i = 0; i < 5; i++) {
            ptrArr[i] = &arr[i];
        }
        ptrToArr = &arr; 
        printf("Accessing elements using the array of pointers:\n");
        for (int i = 0; i < 5; i++) {
            printf("Value at index %d: %d\n", i, *ptrArr[i]);
        }
        printf("\nAccessing elements using the pointer to the array:\n");
        for (int i = 0; i < 5; i++) {
            printf("Value at index %d: %d\n", i, (*ptrToArr)[i]);
        }
    }
OUTPUT:-
Accessing elements using the array of pointers:
Value at index 0: 1
Value at index 1: 2
Value at index 2: 3
Value at index 3: 4
Value at index 4: 5
Accessing elements using the pointer to the array:
Value at index 0: 1
Value at index 1: 2
Value at index 2: 3
Value at index 3: 4
Value at index 4: 5
