2. WAP to add two distance (feet-inch) using the structure.

    #include
    struct feet{
      int foot, inch;
    } d,d1,d2;
    void main(){
      printf("Enter the 1st distance in foot inch format: ");
      scanf("%d%d",&d1.foot,&d1.inch);
      printf("Enter the 2nd distance in foot inch format: ");
      scanf("%d%d",&d2.foot,&d2.inch);
      d.foot = (d1.foot + d2.foot)+(d1.inch+d2.inch)/12;
      d.inch = (d1.inch+d2.inch)%12;
      printf("The sum is %d foot and %d inch.",d.foot,d.inch);
    }
OUTPUT:-
