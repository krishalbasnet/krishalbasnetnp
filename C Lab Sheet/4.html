1. WAP to get your name, address and display using unformatted I/O.

    #include<stdio.h>
    void main(){
      char name[20],add[20];
      printf("Enter your name and address: ");
      gets(name);
      gets(add);
      printf("Name and address are: ");
      puts(name);
      puts(add);
    }
OUTPUT:-
Enter your name and address: John;
Itahari
Name and address are: John
Itahari

2. WAP to get a character form the user using unformatted I/O and display the ASCII value of the entered character.

    #include 
    void main(){
      char c;
      printf("Enter the character: ");
      scanf("%c",&c);
      printf("The ASCII value of %c is %d.",c,c);
    }
OUTPUT:-
Enter the character: a
The ASCII value of a is 97.
