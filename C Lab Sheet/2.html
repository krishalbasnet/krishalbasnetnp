Lab Sheet No.2 - To be familiar Data types, Constants, Operators and Expressions

1. WAP to declare integer, float and character variable. Initialize them with certainvalue and print those values. Also display the size of variables. 
        
    #include<stdio.h>
    void main(){
        int a=2;
        float b=2.6;
        float c='z';
        printf("Integer: %d \t Float: %f \t Character: %c",a,b,c);
        printf("Size of, Int: %d \t float: %d \t char: %d",sizeof(a),sizeof(b),sizeof(c));
    }
OUTPUT:-
Integer: 2  Float: 2.600  Character: z
Size of, Int:4   float:4   char:1

2(a). WAP to swap the values of the variable using third variable. 

    #include<stdio.h>
    void main(){
      int a,b,c;
      printf("Enter 2 numbers: ");
      scanf("%d%d",&a,&b);
      printf("Before swapping: %d, %d\n",a,b);
      c=a;
      a=b;
      b=c;
      printf("After swapping: %d, %d",a,b);
    }
OUTPUT:-
Enter 2 numbers: 2,4
Before swapping: 2,4
After swapping: 4,2

2(b). WAP to swap the values of the variable without using third variable. 

    #include<stdio.h> 
    void main(){
      int x,y;
      printf("Enter 2 numbers: ");
      scanf("%d%d",&x,&y);
      printf("Before swapping: %d, %d\n",x,y);
      x = x+y;
      y = x-y;
      x=x-y
      printf("After swapping: %d, %d",x,y);
    }
Enter 2 numbers: 2,3
Before swapping: 2,3
After swapping: 3,2

3. WAP to calculate the area and volume of a cylinder using pre-processor directive for value of PI. 

    #include<stdio.h>
    #define PI 3.1415
    int main(){
      float r,h,A,V;
      printf("Enter radius and height: ");
      scanf("%f%f",&r,&h);
      A = 2 * PI * r * (r+h);
      V = PI * r * r * h;
      printf("Area: %f, Volume: %f",A,V);
      return 0;
    }
OUTPUT:-
Enter radius and height: 2 3
Area: 62.83, Volume: 37.698

6. What are the output of the following programs:

    #include <stdio.h>
    int main(){
      int a = 5, b = 9;
      printf("a = %d, b = %d\n", a, b);
      printf("a&b = %d\n", a & b);
      printf("a|b = %d\n", a | b);
      printf("a^b = %d\n", a ^ b);
      printf("~a = %d\n", ~a);
      printf("(b<<2)+(a<<1) = %d\n",(b<<2)+(a<<1));
      printf("(b>>1)+(a>>1) = %d\n",(b>>1)+(a>>1));
      return 0;
    }
OUTPUT:-
a=5, b=9
a&b=1
a|b=13
a^b=12
~a=-6
(b<<)+(a<<1)=46
(b>>1)+(a>>1)=6
