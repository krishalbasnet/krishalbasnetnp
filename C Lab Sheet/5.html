5. WAP to display sum of series: 1 + 1/2! + 1/3! + 1/4! + 1/5! ... 1/n!

8. WAP to display weather a number is Armstrong or not. 

    #include<stdio.h>
    #include<math.h>
    void main(){
      int a,b=0,c,n;
      printf("Enter a number: ");
      scanf("%d",&a);
      c=a;
      for(n=0;a!=0;n++){
        a/=10;
      }
      a=c;
      for(a=a;a!=0;a/=10){
        b = b + pow(a%10,n) ;
      }
      if(c==b){
        printf("It is a armstrong number.");
      }
      else{
        printf("It is not an armstrong number.");
      }
    }
OUTPUT:-
Enter a number: 153
It is a armstrong number.

11. WAP to check whether a number is a palindrome or not. 

    #include<stdio.h>
    void main(){
      int num, original, reverse;
      printf("Enter a number: ");
      scanf("%d",&num);
      original = num;
      reverse = 0;
      while(num!-0){
        reverse=reverse*10+(num%10);
        num = num/10;
      }
      if(original == reverse){
        printf("Palindrome");
      }
      else{
        printf("Not a Palindrome");
      }
    }
OUTPUT:-
Enter a number: 12321
Palindrome

12. WAP to find HCF and LCM of two numbers.

    #include<stdio.h>
    void main(){
      int i,num1,num2;
      printf("Enter 2 numbers: ");
      scanf("%d%d",&num1,&num2);
      if(num2>num1){
        for(i=num1;i>0;i--){
          if(num1%i==0 && num2%i ==0){
            printf("The HCF is %d\n",i);
            break;
          }
        }
        for(i=num2;;i+=num2){
          if(i%num1==0 && i%num2 ==0){
            printf("The LCM is %d",i);
          break;
          }
        }
      }
      else{
        printf("Enter the lower number first.");
      }
    }
OUTPUT:-
Enter 2 numbers: 10 12
The HCF is 2
The LCM is 60
