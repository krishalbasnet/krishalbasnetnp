1. WAP to create and store string using in a text file. Also display the string from the text file.

    #include<stdio.h>
    void main(){
        FILE *fp;
        char str[50];
        fp = fopen("newfile.txt","w");
        printf("Enter a string: ");
        scanf("%s",str);
        fprintf(fp,"%s",str);
        fclose(fp);
        fp = fopen("newfile.txt","r");
        fscanf(fp,"%s",str);
        fclose(fp);
        printf("The string is %s",str);
    }
OUTPUT:-
Enter a string: NEPAL
The string is NEPAL

2. WAP to illustrate the use of single character input/output in file (fgetc() and fputc()).

    #include<stdio.h>
    void main(){
        FILE *fp;
        char ch;
        fp = fopen("newfile.txt","w");
        printf("Enter a chararcter: ");
        scanf("%c",&ch);
        fputc(ch,fp);
        fclose(fp);
        fp = fopen("newfile.txt","r");
        ch = fgetc(fp);
        printf("You enetered %c",ch);
        fclose(fp);
    }
OUTPUT:-
Enter a chararcter: k
You enetered k
