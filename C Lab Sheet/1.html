Lab Sheet No.1 - To be familiar with C-Programming

1. WAP to display hello world. 
        
    #include<stdio.h>
    void main(){
       printf("Hello World");
    }
OUTPUT:-
Hello World

2. WAP to display your name, roll number and address. 
            
    #include<stdio.h>
    void main(){
        printf("Name: John Smith");
        printf("Roll no.: 90");
        printf("Address: London");
    }
OUTPUT:-
Name: John Smith
Roll no.: 90
Address: London

3. WAP to add two integer variables and print sum. 
              
    #include <stdio.h>
    void main(){
      int a,b,sum;
      printf("Enter two numbers: ");
      scanf("%d%d", &a, &b);
      sum = a+b;
      printf("The sum is %d",sum);
    }
OUTPUT:-
Enter two numbers: 2 3
The sum is 5

4. WAP to multiply two integer variables and print product. 
              
    #include <stdio.h>
    void main(){
      int a,b,product;
      printf("Enter two numbers: ");
      scanf("%d%d", &a, &b);
      product = a*b;
      printf("The product is %d",product);
    }
OUTPUT:-
Enter two numbers: 2 3
The product is 6

5. WAP to calculate and display the simple interest. 

    #include
    void main(){
        float p,t,r,SI;
        printf("Enter Principle, Time and Rate: ");
        scanf("%f%f%f",&p,&t,&r);
        SI = (p*t*r)/100;
        printf("The Simple Interest is %f",SI);
    }
OUTPUT:-
Enter Principle, Time and Rate: 2000 3 4
The Simple Interest is 240.000000

6. WAP to calculate the area of the circle. 

    #include
    #define pi 3.1415
    void main(){
        float rad, area;
        printf("Enter the radius: ");   
        scanf("%f",&rad);
        area = pi * rad * rad;
        printf("The area is %f",area);
    }
OUTPUT:-
Enter the radius: 7
The area is 153.933502
