3. WAP to count the length of string without using string function.

    #include<stdio.h>
    void main(){
      char str[20],i;
      printf("Enter the string:");
      scanf("%s",str);
      i=0;
      while(str[i]!='\0'){
        i++;
      }
      printf("The length is %d",i);
    }
OUTPUT:-
Enter the string:Programmer
The length is 10

5. WAP to compare two string without using string function.

    #include<stdio.h>
    int compareStrings(char* str1, char* str2) {
        int i = 0;
        while (str1[i] != '\0' && str2[i] != '\0') {
            if (str1[i] != str2[i]) {
                return 0;
            }
            i++;
        }
        if (str1[i] == '\0' && str2[i] == '\0') {
            return 1;
        }
        return 0;
    }
    void main() {
        char string1[100], string2[100];
        printf("Enter the first string: ");
        scanf("%s", string1);
        printf("Enter the second string: ");
        scanf("%s", string2);
        if (compareStrings(string1, string2)) {
            printf("The strings are equal.\n");
        } else {
            printf("The strings are not equal.\n");
        }
    }
OUTPUT:-
Enter the first string: man
Enter the second string: woman
The strings are not equal.

6. WAP to concatenate two string using user define function.

    #include<stdio.h>
    void concatStrings(char* str1, char* str2, char* result) {
        int i = 0, j = 0;
        while (str1[i] != '\0') {
            result[j] = str1[i];
            i++;
            j++;
        }
        i = 0;
        while (str2[i] != '\0') {
            result[j] = str2[i];
            i++;
            j++;
        }
        result[j] = '\0';
    }
    void main() {
        char string1[100], string2[100], result[200];
        printf("Enter the first string: ");
        scanf("%s", string1);
        printf("Enter the second string: ");
        scanf("%s", string2);
        concatStrings(string1, string2, result);
        printf("Concatenated string: %s\n", result);
    }
OUTPUT:-
Enter the first string: hello
Enter the second string: world
Concatenated string: helloworld

8. WAP to check whether the string is palindrome or not.

    #include<stdio.h>
    #include<string.h>
    void main(){
      char original[20], reverse[20];
      printf("Enter the string: ");
      scanf("%s",original);
      strcpy(reverse,original);
      strrev(reverse);
      if(strcmp(original,reverse)==0){
        printf("It is a palindrome.");
      }
      else{
        printf("It is not a palindrome.");
      }
    }
OUTPUT:-
Enter the string: madam
It is a palindrome.
