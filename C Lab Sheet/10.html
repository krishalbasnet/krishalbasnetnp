8. WAP to check whether the string is palindrome or not.

    #include<stdio.h>
    #include<string.h>
    void main(){
      char original[20], reverse[20];
      printf("Enter the string: ");
      scanf("%s",original);
      strcpy(reverse,original);
      strrev(reverse);
      if(strcmp(original,reverse)==0){
        printf("It is a palindrome.");
      }
      else{
        printf("It is not a palindrome.");
      }
    }
OUTPUT:-
Enter the string: madam
It is a palindrome.
