3. WAP to calculate factorial using Recursion.

    #include<stdio.h>
    int factorial(int x);
    void main(){
      int n;
      printf("Enter the number: ");
      scanf("%d",&n);
      printf("The factorial of %d is %d.",n,factorial(n));
    }
    int factorial(int x){
      int fact=1 ;
      if(x>1){
        fact = x * factorial(x-1);
      }else{
        return fact;
      }
    }
OUTPUT:-
Enter the number: 4
The factorial of 4 is 24.

4. WAP to display the nth Fibonacci number using recursion.

    #include<stdio.h>
    int fibonacci(int x);
    void main(){
      int n;
      printf("Enter the number: ");
      scanf("%d",&n);
      printf("The fibonacci of %d is %d.",n,fibonacci(n));
    }
    int fibonacci(int x){
      int fib=0 ;
      if(x>0){
        fib = x + fibonacci(x-1);
      }else{
        return fib;
      }
    }
OUTPUT:-
Enter the number: 5
The fibonacci of 5 is 15.
