1. Write a program to add, subtract, multiply and divide two integers using user defined function add(), sub(), mul() and div().

    #include<stdio.h>
    int add(int x,int y);
    int mul(int x,int y);
    int sub(int x,int y);
    float div(int x,int y);
    void main(){
      int a,b,op;
      printf("Enter two numbers: ");
      scanf("%d%d",&a,&b);
      printf("Enter the operator as follow:\n1.Add\n2.Substraction\n3.Multiplication\n4.Divide\n");
      scanf("%d",&op);
      switch(op){
        case 1: printf("The sum is %d",add(a,b));
                break;
        case 2: printf("The difference is %d",sub(a,b));
                break;
        case 3: printf("The product is %d",mul(a,b));
                break;
        case 4: printf("The dividend is %f",div(a,b));
                break;
        default:printf("Error");
      }
    }
    int add(int x,int y){
      return x+y;
    }
    int sub(int x,int y){
      return x-y;
    }
    int mul(int x,int y){
      return x*y;
    }
    float div(int x,int y){
      return (float)x/y;
    }

1.Add

2. WAP to display sum of series: x + x2 /2! + x3 /3! + x4 /4! + x5 /5! ... xn /n!. User defined function factorial() and power() should be used to calculate the factorial and power.

    #include <stdio.h>
    int power(int m, int n);
    void main(){
      int n, fact = 1, i, j, x;
      float sum;
      printf("Enter the value of n: ");
      scanf("%d", &n);
      printf("Enter the value of x: ");
      scanf("%d", &x);
      sum = 0.00;
      for (i = 1; i <= n; i++){
        fact = 1.00;
        for (j = i; j > 0; j--){
          fact = fact * j;
        }
        sum = sum + (power(x, i) / (float)fact);
      }
      printf("The sum is: %f", sum);
    }
    int power(int m, int n){
      int i,out=1;
      for(i=n;i>0;i--){
        out = out * m;
      }
      return out;
    }
OUTPUT:-
Enter the value of n: 5
Enter the value of x: 2
The sum is: 6.266667

4. WAP to display the nth Fibonacci number using recursion.

    #include<stdio.h>
    int fibonacci(int x);
    void main(){
        int n;
        printf("Enter the number: ");
        scanf("%d", &n);
        printf("The %dth fibonacci number is %d.", n, fibonacci(n));
    }
    int fibonacci(int x){
        int fib = 0;
        if(x==1){
            return 1;
        }
        else if(x==2){
            return 1;
        }
        else if (x > 0){
            fib = fibonacci(x-2) + fibonacci(x - 1);
        }
        else{
            return fib;
        }
    }
OUTPUT:-
Enter the number: 5
The fibonacci of 5 is 15.

5. WAP to take two numbers in main(). Write a function Swap() to swap the values of the variables. Print the swapped values in main().

    #include<stdio.h>
    void Swap(int *a, int *b);
    int main() {
      int a, b;
      printf("Enter the first number: ");
      scanf("%d", &a);
      printf("Enter the second number: ");
      scanf("%d", &b);
      printf("Before swapping: num1 = %d, num2 = %d\n", a, b);
      Swap(&a, &b);
      printf("After swapping: num1 = %d, num2 = %d\n", a, b);
      return 0;
    }
    void Swap(int *a, int *b) {
      int temp = *a;
      *a = *b;
      *b = temp;
    }
OUTPUT:-
Enter the first number: 10
Enter the second number:20
Before swapping: num1 = 10, num2 = 20
After swapping: num1 = 20, num2 = 10
