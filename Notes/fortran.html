PADHA VAI PADHA

Structure of Fortran:-
program program_name code ..... end program program_name

Q. WAP to display Hello World:-

program myFirstprg implicit none print *,"Hello World" end program myFirstprg

Q. WAP to add two numbers:-

program myCode implicit none integer :: x,y,z read *,x,y z=x+y print *,z end program myCode

Q. WAP to check positive or negative number:-

program myCode implicit none integer :: x read *,x if (x>0) then print *,"Positive" else if(x<0) then print *,"Negative" else print *,"Zero" end if end program myCode

Q. WAP to display the greatest among three numbers:-

program myCode implicit none integer :: x,y,z read *,x,y,z if (x>y .AND. x>z) then print *,x else if(y>z) then print *,y else print *,z end if end program myCode

Q. WAP to display 1 to 10 using loop:-

program myCode implicit none integer :: i do i=1,10,1 print *,i end do end program myCode

Q. WAP to check Prime number:-

program code integer :: x,i read *,x if (x == 0) then print *,"You entered zero." else if (x == 1) then print *,"You entered one." else if (x == 2) then print *,"Smallest and only even Prime number." else do i=2,x-1 if (MOD(x,i)==0) then print *,"Non-Prime" GOTO 55 end if end do print *,"Prime" end if 55 end program code

Q. WAP to check Palindrome:-

program Palindrome implicit none integer :: num, original, reverse=0 print *,"Enter the number: " read *,num original =num do while (num /= 0) reverse = reverse *10 + MOD(num,10) num = num/10 end do if(reverse .EQ. original) then print *,"Palindrome" else print *,"Not Palindrome" end if end program Palindrome
